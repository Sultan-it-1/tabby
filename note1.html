// --- منطق الاستيراد المطور (حل مشكلة التكرار) --- function importData(e) { const file = e.target.files[0]; if (!file) return; const reader = new FileReader(); reader.onload = (x) => { try { const importedData = JSON.parse(x.target.result); // المرور على اللغات (ar, en) ['ar', 'en'].forEach(lang => { if (importedData[lang]) { for (let cat in importedData[lang]) { // إذا كان القسم غير موجود، ننشئه if (!storageData[lang][cat]) { storageData[lang][cat] = []; } // دمج النوتات داخل القسم importedData[lang][cat].forEach(newItem => { // فحص هل الاسم موجود مسبقاً في هذا القسم let finalLabel = newItem.l; const existingLabels = storageData[lang][cat].map(item => item.l); let counter = 1; while (existingLabels.includes(finalLabel)) { const suffix = lang === 'ar' ? `(نسخة ${counter})` : `(Copy ${counter})`; finalLabel = `${newItem.l} ${suffix}`; counter++; } // إضافة النوتة بمعرف جديد واسم (ربما معدل) storageData[lang][cat].push({ id: Date.now() + Math.random(), // معرف فريد جداً l: finalLabel, t: newItem.t }); }); } } }); saveAndRefresh(); showStatus("تم الدمج بنجاح! ✅"); e.target.value = ""; // تفريغ المدخل ليتيح استيراد نفس الملف مرة أخرى } catch (err) { alert("حدث خطأ في قراءة الملف، تأكد من جودة ملف الـ JSON"); console.error(err); } }; reader.readAsText(file); }
